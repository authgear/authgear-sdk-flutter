bool get canReauthenticate {
  final idToken = _idToken;
  if (idToken == null) {
    return false;
  }
  final payload = decodeIDToken(idToken);
  final can = payload["https://authgear.com/claims/user/can_reauthenticate"];
  return can is bool && can == true;
}
